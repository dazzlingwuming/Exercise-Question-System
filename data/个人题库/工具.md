  2. `properties` 中字段是否全部出现在 `required`

reference_answer: |
  核心思路是递归遍历每个工具的参数 schema，遇到 object 就做两类校验：一类是 closed schema 校验，即 `additionalProperties` 是否为 `false`；另一类是 required completeness 校验，即 `properties` 里的键是否全部在 `required` 中。若发现“字段存在于 properties，但不在 required”，则在 strict mode 下应报出问题，并在提示中给出“如确需可选，请考虑使用 `null` 联合类型”的修复建议。这个思路与 OpenAI 文档对 strict mode 的要求一致。citeturn12view0turn13view0

  实现时还应注意数组、嵌套对象和组合类型。对于 `type: ["string", "null"]` 这类字段，不代表它不需要出现在 `required`；相反，strict mode 下它依然应该被列入 required，只是值可以取 null。citeturn13view0

  - required 完整性
  - null 联合

scoring_points:

  - 能覆盖 object 和嵌套 object
  - 能检查 additionalProperties 与 required 两项核心规则
  - 能对“可选字段”给出合理修复建议
  - 能考虑路径输出，方便定位错误字段

pseudocode: |
  function validate_strict_tools(tools):
      issues = []

      for tool in tools:
          schema = get_tool_parameters(tool)
          walk_schema(
              schema=schema,
              path=["tools", tool.name, "parameters"],
              issues=issues
          )

      return issues

  function walk_schema(schema, path, issues):
      if schema is null:
          return

      schema_type = schema.get("type")

      if is_object_type(schema_type):
          props = schema.get("properties", {})
          required = set(schema.get("required", []))

          if schema.get("additionalProperties") is not false:
              issues.append({
                  "path": join(path),
                  "error": "missing_additionalProperties_false"
              })

          for key, child in props.items():
              if key not in required:
                  issues.append({
                      "path": join(path + ["properties", key]),
                      "error": "property_not_required_under_strict",
                      "suggestion": "keep field in required; if logically optional, use union with null"
                  })
              walk_schema(child, path + ["properties", key], issues)

      if schema_type == "array" and "items" in schema:
          walk_schema(schema["items"], path + ["items"], issues)

      for combiner in ["allOf", "anyOf", "oneOf"]:
          if combiner in schema:
              for idx, child in enumerate(schema[combiner]):
                  walk_schema(child, path + [combiner, str(idx)], issues)

  function is_object_type(schema_type):
      if schema_type == "object":
          return true
      if is_list(schema_type) and "object" in schema_type:
          return true
      return false

complexity:
  time: O(N)，其中 N 为 schema 节点总数
  space: O(H + M)，其中 H 为递归深度，M 为问题列表大小

  function dispatch_tool_calls(tool_calls, registry, request_context):
      outputs = []

      for call in tool_calls:
          result = handle_one_call(call, registry, request_context)
          outputs.append({
              "type": "function_call_output",
              "call_id": call.call_id,
              "output": to_json_string(result)
          })

      return outputs

  function handle_one_call(call, registry, request_context):
      if call.name not in registry:
          return normalize_error(
              kind="unknown_tool",
              retryable=false,
              message="tool not registered",
              details={"tool_name": call.name}
          )

      tool_meta = registry[call.name]

      try:
          args = json_parse(call.arguments)
      except Exception as e:
          return normalize_error(
              kind="invalid_arguments_json",
              retryable=false,
              message="arguments JSON parse failed",
              details={"raw_arguments": call.arguments, "error": str(e)}
          )

      validation_error = tool_meta.validator(args)
      if validation_error is not null:
          return normalize_error(
              kind="business_validation_failed",
              retryable=false,
              message="arguments failed business validation",
              details=validation_error
          )

      if tool_meta.is_side_effect:
          idem_key = build_idempotency_key(
              user_id=request_context.user_id,
              tool_name=call.name,
              args=args
          )

          existing = read_execution_record(idem_key)
          if existing and existing.status == "succeeded":
              return {
                  "status": "ok",
                  "source": "idempotent_replay",
                  "data": existing.result
              }

          if existing and existing.status == "running":
              return normalize_error(
                  kind="execution_in_progress",
                  retryable=true,
                  message="same idempotent action is running",
                  details={"idempotency_key": idem_key}
              )

          save_execution_record(idem_key, status="running")

          try:
              data = tool_meta.callable(**args)
              save_execution_record(idem_key, status="succeeded", result=data)
              return {"status": "ok", "source": "fresh_execution", "data": data}
          except Exception as e:
              save_execution_record(idem_key, status="failed", error=str(e))
              return normalize_error(
                  kind="side_effect_execution_failed",
                  retryable=false,
                  message="side-effect tool failed",
                  details={"idempotency_key": idem_key, "error": str(e)}
              )

      else:
          attempts = 0
          last_error = null
          while attempts <= tool_meta.retry_policy.max_retries:
              try:
                  data = tool_meta.callable(**args)
                  return {"status": "ok", "source": "read_only_execution", "data": data}
              except RetryableError as e:
                  last_error = e
                  attempts += 1
                  sleep(backoff(attempts))
              except Exception as e:
                  return normalize_error(
                      kind="read_only_execution_failed",
                      retryable=false,
                      message="read-only tool failed with non-retryable error",
                      details={"error": str(e)}
                  )

          return normalize_error(
              kind="read_only_retry_exhausted",
              retryable=true,
              message="read-only retries exhausted",
              details={"error": str(last_error), "attempts": attempts}
          )

  function normalize_error(kind, retryable, message, details):
      return {
          "status": "error",
          "error_type": kind,
          "retryable": retryable,
          "message": message,
          "details": details
      }
